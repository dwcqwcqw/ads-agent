import os
import json
from typing import TypedDict, Dict, Any, List
from langgraph.graph import StateGraph, END
from openai import OpenAI

client = OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

# =========================
# 状态定义
# =========================
class State(TypedDict):
    raw: Dict[str, Any]
    enriched: Dict[str, Any]
    profile: Dict[str, Any]
    matched_ads: List[Dict[str, Any]]
    creative: Dict[str, Any]
    image_path: str
    sales: Dict[str, Any]
    conversion: Dict[str, Any]
    optimize: Dict[str, Any]


# =========================
# 工具函数
# =========================
def ask_json(prompt):
    res = client.chat.completions.create(
        model="gpt-5.5",
        messages=[{"role": "user", "content": prompt}],
        temperature=1.0
    )
    txt = res.choices[0].message.content.strip()
    txt = txt.replace("```json", "").replace("```", "")
    try:
        return json.loads(txt)
    except:
        return {"raw": txt}


# =========================
# Agent 1：用户画像
# =========================
def agent_profile(state: State):
    prompt = f"""
基于小学数学学习数据生成用户画像：

{state["raw"]}

输出JSON：
认知水平、核心问题、需求、转化意向
"""
    state["profile"] = ask_json(prompt)
    return state


# =========================
# Agent 2：数据补全（手机号）
# =========================
def agent_enrich(state: State):
    external = {
        "购物": ["儿童学习机", "益智玩具"],
        "APP": ["抖音", "作业帮"],
        "社媒": ["亲子教育"],
        "消费能力": "中高"
    }

    prompt = f"""
基于手机号补全数据：

用户：
{state["profile"]}

外部：
{external}

输出增强画像
"""
    state["enriched"] = ask_json(prompt)
    return state


# =========================
# Agent 3：广告匹配（跨行业）
# =========================
def agent_match(state: State):
    ads = [
        {"行业": "教育", "产品": "数学提分班"},
        {"行业": "电商", "产品": "学习机"},
        {"行业": "内容", "产品": "AI学习会员"}
    ]

    prompt = f"""
匹配最适合广告：

用户：
{state["enriched"]}

广告池：
{ads}

输出Top3
"""
    state["matched_ads"] = ask_json(prompt).get("Top3", ads)
    return state


# =========================
# Agent 4：广告生成（含真实图片）
# =========================
def agent_creative(state: State):

    prompt = f"""
生成广告素材：

广告：
{state["matched_ads"]}

输出：
标题、文案、图片描述
"""
    creative = ask_json(prompt)

    # 生成图片（真实）
    image = client.images.generate(
        model="gpt-image-1.5",
        prompt=creative.get("图片描述", "小学生学习数学场景"),
        size="1024x1024"
    )

    image_bytes = image.data[0].b64_json
    image_path = "ad_image.png"

    import base64
    with open(image_path, "wb") as f:
        f.write(base64.b64decode(image_bytes))

    state["creative"] = creative
    state["image_path"] = image_path
    return state


# =========================
# Agent 5：AI销售
# =========================
def agent_sales(state: State):
    prompt = f"""
模拟销售对话：

广告：
{state["creative"]}

输出对话+成交率
"""
    state["sales"] = ask_json(prompt)
    return state


# =========================
# Agent 6：优化分析
# =========================
def agent_optimize(state: State):

    conversion = {
        "曝光": 1000,
        "点击": 200,
        "注册": 80,
        "付费": 20
    }

    prompt = f"""
分析漏斗并优化：

{conversion}

输出优化建议
"""
    state["conversion"] = conversion
    state["optimize"] = ask_json(prompt)
    return state


# =========================
# 构建 LangGraph
# =========================
graph = StateGraph(State)

graph.add_node("profile", agent_profile)
graph.add_node("enrich", agent_enrich)
graph.add_node("match", agent_match)
graph.add_node("creative", agent_creative)
graph.add_node("sales", agent_sales)
graph.add_node("optimize", agent_optimize)

graph.set_entry_point("profile")

graph.add_edge("profile", "enrich")
graph.add_edge("enrich", "match")
graph.add_edge("match", "creative")
graph.add_edge("creative", "sales")
graph.add_edge("sales", "optimize")
graph.add_edge("optimize", END)

app = graph.compile()


# =========================
# 运行
# =========================
if __name__ == "__main__":

    state = {
        "raw": {
            "phone": "138****1234",
            "grade": "小学五年级",
            "weak": ["应用题", "分数"],
            "dialogue": ["孩子不会应用题"]
        }
    }

    result = app.invoke(state)

    print("\n==== 用户画像 ====")
    print(json.dumps(result["profile"], ensure_ascii=False, indent=2))

    print("\n==== 广告文案 ====")
    print(json.dumps(result["creative"], ensure_ascii=False, indent=2))

    print("\n==== 广告图片保存路径 ====")
    print(result["image_path"])

    print("\n==== 转化优化 ====")
    print(json.dumps(result["optimize"], ensure_ascii=False, indent=2))
